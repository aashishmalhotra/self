class Solution:
    def majorityElement(self, nums: List[int]) -> int:
        maxcount = 0
        index = -1
        n = len(nums)

        for i in range(n):
            count = 1
            for j in range(i+1,n):
                if nums[i] == nums[j]:
                    count+=1
        # Store maxcount of nums
            if count > maxcount:
                maxcount = count
                index = i

        # Check maxcount is greater than n/2
        if maxcount>n/2:
            return nums[index]
        return index
