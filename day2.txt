class Solution:
    def maxProfit(self, prices: List[int]) -> int:
        l , r = 0 , 1
        maxP = 0
        n = 0
        while r < len(prices):
            if prices[l] < prices[r]:
                profit = prices[r] - prices[l]
                # print(profit)
                maxP = max(profit,maxP)
                # print(maxP)
                # r+=1
            else:
                # found right price low then left , so l = r (buy should always be low)
                l = r
            r +=1

        return maxP
